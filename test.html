namespace BeatsWorking
{
	public partial class GridViewPager : System.Web.UI.UserControl
	{
		private GridView _gridView;

		protected void Page_Load(object sender, EventArgs e)
		{
			Control c = Parent;
			while (c != null)
			{
				if (c is GridView)
				{
					_gridView = (GridView)c;
					break;
				}
				c = c.Parent;
			}
		}

		protected void TextBoxPage_TextChanged(object sender, EventArgs e)
		{
			if (_gridView == null)
			{
				return;
			}
			int page;
			if (int.TryParse(TextBoxPage.Text.Trim(), out page))
			{
				if (page <= 0)
				{
					page = 1;
				}
				if (page > _gridView.PageCount)
				{
					page = _gridView.PageCount;
				}
				_gridView.PageIndex = page - 1;
			}
			TextBoxPage.Text = (_gridView.PageIndex + 1).ToString();
		}

		protected void DropDownListPageSize_SelectedIndexChanged(object sender, EventArgs e)
		{
			if (_gridView == null)
			{
				return;
			}
			DropDownList dropdownlistpagersize = (DropDownList)sender;
			_gridView.PageSize = Convert.ToInt32(dropdownlistpagersize.SelectedValue);
			int pageindex = _gridView.PageIndex;
			_gridView.DataBind();
			if (_gridView.PageIndex != pageindex)
			{
				//if page index changed it means the previous page was not valid and was adjusted. Rebind to fill control with adjusted page
				_gridView.DataBind();
			}
		}

		protected void Page_PreRender(object sender, EventArgs e)
		{
			if (_gridView != null)
			{
				LabelNumberOfPages.Text = _gridView.PageCount.ToString();
				TextBoxPage.Text = (_gridView.PageIndex + 1).ToString();
				DropDownListPageSize.SelectedValue = _gridView.PageSize.ToString();
			}
		}
	}
}

Public Class BugManager
   ' Do not modify this class.

   Private Shared _bugs As New List(Of Bug)()
   Private Shared _nextBugId As Integer = 1

   Private Sub New()
   End Sub

   Shared Sub New()
      InitializeBugData()
   End Sub

    _
   Public Shared Function GetAllBugs() As IEnumerable(Of Bug)
      Return _bugs
   End Function

    _
   Public Shared Sub UpdateBug(ByVal bug As Bug)

      Dim context As New FindContext(bug)
      Dim bugInList As Bug = _bugs.Find(AddressOf context.IsMatch)

      If bugInList Is Nothing Then
         Return
      End If

      bugInList.Description = bug.Description
      bugInList.AssignedTo = bug.AssignedTo
      bugInList.Status = bug.Status
   End Sub

   Private Shared Sub InitializeBugData()
      AddNewBug("Error occured when Submit button is clicked.", "Tom")
      AddNewBug("Would like to see the ID when a new item is added.", "Karen")
      AddNewBug("Difficult to read yellow font on home page.", "Tom")
      AddNewBug("Page not found error occurred when the link on the home " & "page was clicked.", "Karen")
   End Sub

   Private Shared Sub AddNewBug(ByVal bugDescription As String, ByVal reportedBy As String)
      _nextBugId += 1
      Dim id As Integer = _nextBugId
      _bugs.Add(New Bug(id, bugDescription, BugStatus.Open, reportedBy))
   End Sub

   Private Class FindContext

      Private _bug As Bug

      Public Sub New(ByVal bug As Bug)
         _bug = bug
      End Sub

      Public Function IsMatch(ByVal bug As Bug) As Boolean
         Return (_bug.Id = bug.Id)
      End Function
   End Class

End Class
